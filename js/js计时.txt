	<script type="text/javascript">
		var c=0
		var t
		function timedCount()
		{
			document.getElementById('txt').value=c
			c=c+1
			t=setTimeout("timedCount()",1000)
		}
		function stopCount()
		{
		c=0;
		document.getElementById('txt').value=c;
		clearTimeout(t);
		}
	</script>

	function startTime()
